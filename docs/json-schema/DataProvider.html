DataProvider

Type: object

A vizzToolsCore DataProvider definition object that describes the information needed for accessing the dataset data items.
Examples:

{
    "type": "coverage",
    "name": "rasterio",
    "data": "tests/data/CMC_glb_TMP_TGL_2_latlon.15x.15_2020081000_P000.grib2",
    "options": {
        "DATA_ENCODING": "COMPLEX_PACKING"
    },
    "format": {
        "name": "GRIB",
        "mimetype": "application/x-grib2"
    }
}

{
    "type": "coverage",
    "name": "xarray",
    "data": "tests/data/coads_sst.nc",
    "x_field": "lon",
    "time_field": "time",
    "format": {
        "name": "netcdf",
        "mimetype": "application/x-netcdf"
    }
}

{
    "type": "coverage",
    "name": "xarray",
    "data": "tests/data/analysed_sst.zarr",
    "format": {
        "name": "zarr",
        "mimetype": "application/zip"
    }
}
